// วาง code นี้ใน Google Apps Script แล้ว Deploy เป็น Web App // Execute as: Me | Who has access: Anyone const SHEET_KEYS = 'Keys'; const SHEET_MSGS = 'Messages'; function doGet(e) { const a = e.parameter.action; if (a === 'getKeys') return getKeys(); if (a === 'getMsgs') return getMsgs(); return resp({ok:true, msg:'PATO GAS ONLINE'}); } function doPost(e) { const d = JSON.parse(e.postData.contents); if (d.action === 'addKey') return addKey(d.key); if (d.action === 'deleteKey') return deleteKey(d.key); if (d.action === 'addMsg') return addMsg(d.msg); return resp({ok:false}); } function getKeys() { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_KEYS); const rows = sh.getDataRange().getValues().slice(1); return resp({ok:true, keys: rows.map(r => ({key:r[0],permanent:r[1],expireAt:r[2],createdAt:r[3]}))}); } function addKey(k) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_KEYS); sh.appendRow([k.key, k.permanent, k.expireAt || '', k.createdAt]); return resp({ok:true}); } function deleteKey(keyVal) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_KEYS); const data = sh.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (data[i][0] === keyVal) sh.deleteRow(i + 1); } return resp({ok:true}); } function addMsg(m) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_MSGS); sh.appendRow([m.to, m.msg, m.sentAt]); return resp({ok:true}); } function getMsgs() { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_MSGS); const rows = sh.getDataRange().getValues().slice(1); return resp({ok:true, msgs: rows.map(r => ({to:r[0],msg:r[1],sentAt:r[2]})).reverse()}); } function resp(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
